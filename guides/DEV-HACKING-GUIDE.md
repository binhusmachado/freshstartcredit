### Example 1: Automated Lead Scraping

```python
# Robot creates this script:

"""
Daily Lead Generation Bot
Scrapes LinkedIn for credit repair prospects
"""

import schedule
import time
from linkedin_scraper import scrape_profiles
from openai import OpenAI

client = OpenAI()

def find_leads():
    # Search for people with bad credit indicators
    search_terms = [
        "mortgage denied",
        "credit repair help",
        "fix my credit",
        "credit score low"
    ]
    
    leads = []
    for term in search_terms:
        results = scrape_profiles(term, limit=10)
        leads.extend(results)
    
    # Score leads with AI
    for lead in leads:
        score = client.chat.completions.create(
            model="gpt-4",
            messages=[{
                "role": "user",
                "content": f"Score this lead 1-10: {lead}"
            }]
        )
        lead['score'] = score.choices[0].message.content
    
    # Save to Airtable
    save_to_airtable(leads)
    
    # Send alert
    send_telegram(f"Found {len(leads)} new leads today!")

# Run daily at 9am
schedule.every().day.at("09:00").do(find_leads)

while True:
    schedule.run_pending()
    time.sleep(60)
```
